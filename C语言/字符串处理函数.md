### 字符串处理函数

* 字符串复制函数： strcpy()
```c
#include <stdio.h>
#include <string.h>

int main()
{
    char name[20],nameNew[20];

    printf("请输入您的姓名：\n");
    scanf("%s", name);

    strcpy(nameNew, name);

    printf("Hello,%s\n", nameNew);

    return 0;
}

```

* 字符串拼接函数： strcat()
```c
#include <stdio.h>
#include <string.h>

int main()
{
    char name[30],address[300],info[330];

    printf("请输入您的姓名：\n");
    scanf("%s", name);

    printf("请输入您的家庭地址：\n");
    scanf("%s", address);

    strcat(info, name);
    strcat(info, ",");
    strcat(info, address);

    printf("您的个人信息是：%s\n", info);

    return 0;
}

```

* 字符串比较函数： strcmp()
```c
#include <stdio.h>
#include <string.h>

int main()
{
    char student1[30] = "zisi", student2[30] = "lisi";
    int result;
    
    printf("请输入学生一的姓名：\n");
    scanf("%s", student1);
    printf("请输入学生二的姓名：\n");
    scanf("%s", student2);

    result = strcmp(student1, student2);
    printf("%d\n", strcmp(student1, student2));

    if (result > 0) {
        printf("学生1的姓名排在学生2的前面\n");
    } else if (result == 0) {
        printf("学生1的姓名和学生2的姓名不分先后\n");
    } else {
        printf("学生1的姓名排在学生2的后面\n");
    }   

    return 0;
}

```
