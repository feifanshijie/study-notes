### 一些习题

* 输入一个正整数，判断它有多少位
```c
#include <stdio.h>

int main()
{
    int x = 0;
    int n = 0;
    
    printf("请输入一个正整数：");
    scanf("%d", &x);
    
    while (x > 0) {
        n++;
        x /= 10;
    }
    printf("您输入的是一个%d位数", n);
    
    return 0;
}
```

* 计算1+1/2+1/3+...+1/n的和
```c
#include <stdio.h>

int main()
{
    double sum = 0.0;
    int n = 0;
    int i = 0;
    
    printf("请输入一个正整数：");
    scanf("%d", &n);
    
    for (i = 1; i <= n; i++) {
        sum += 1.0 / i;
    }
    printf("和为：%f", sum);
    
    return 0;
}
```

* 计算1-1/2+1/3-1/4...+1/n的和
```c
#include <stdio.h>

int main()
{
    double sum = 0.0;
    int n = 0;
    int i = 0;
    double sign = 1.0;
    
    printf("请输入一个正整数：");
    scanf("%d", &n);
    
    for (i = 1; i <= n; i++) {
        sum += sign / i;
        sign = -sign;
    }
    printf("和为：%f", sum);
    
    return 0;
}
```
