### 一些习题

* 输入一个正整数，判断它有多少位
```c
#include <stdio.h>

int main()
{
    int x = 0;
    int n = 0;
    
    printf("请输入一个正整数：");
    scanf("%d", &x);
    
    while (x > 0) {
        n++;
        x /= 10;
    }
    printf("您输入的是一个%d位数", n);
    
    return 0;
}
```

* 计算1+1/2+1/3+...+1/n的和
```c
#include <stdio.h>

int main()
{
    double sum = 0.0;
    int n = 0;
    int i = 0;
    
    printf("请输入一个正整数：");
    scanf("%d", &n);
    
    for (i = 1; i <= n; i++) {
        sum += 1.0 / i;
    }
    printf("和为：%f", sum);
    
    return 0;
}
```

* 计算1-1/2+1/3-1/4...+1/(n-1)-1/n的和
```c
#include <stdio.h>

{

    for (i = 1; i <= n; i++) {
        sum += sign / i;
        sign = -sign;

    printf("和为：%f", sum);
    
    return 0;
}
```

* 使用for循环显示10个随机数字
```c
#include <stdio.h>
#include <time.h>
#include <stdlib.h>

int main(void)
{
    srand(time(NULL)); /* 以当前时间作为随机种子 */
    for (int i = 0; i < 10; i++) {
        /* 如果在使用rand()函数之前不提供种子，也就是不用srand()函数设定种子值，
        则rand()函数总是默认以1作为种子，程序每次运行将产生同样的随机数序列 */ 
        printf("第%d次产生的随机数是：%d\n", i+1, rand());
    }
    
    return 0;
}
```

* 请输入一个正整数n，然后我们会输出n行金字塔图案：
```c
#include <stdio.h>

int main(void)
{
	/* 
	 * 输出n行金字塔型图案: 可以想象成先显示一个倒立的直角三角形（由空格组成），然后再输出一个正立的三角形
	 */
	int n = 0;
	printf("请输入一个正整数n，然后我们会输出n行金字塔图案：");
	scanf("%d", &n);
	
	for (int i = 1; i <= n; i++) {
		for (int j = 0; j < (n-i); j++) {
			printf(" ");
		}
		for (int k = 0; k < 2*i-1; k++) {
			printf("*"); 
		}
		printf("\n");
	}
	
	return 0;
}
```

* 输出九九乘法表
```c
#include <stdio.h>

int main(void)
{
	/* 输出九九乘法表 */
	for (int i = 1; i <= 9; i++) {
		for (int j = 1; j <= i; j++) {
			printf("%d*%d=%d ", i, j, i*j);
		}
		printf("\n");
	}
	
	return 0;
}
```

* 将用户输入的一个大写字母转成对应的小写字母输出
```c
#include <stdio.h>

int main(void)
{
	/* 将用户输入的一个大写字母转成对应的小写字母输出 */
	char big;
	char small;
	
	puts("请输入一个大写字母：");
	big = getchar();
	
	puts("对应的小写字母为：");
	/*
	 * 大写字母与小写字母的ASCII码值相差32。
	 * 例如字符A的ASCII码值为65，则a的ASCII码值为97。 
	 */ 
	small = big + 32;
	putchar(small);
	
	return 0;
}
```
